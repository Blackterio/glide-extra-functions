
# This is used mainly for toggleable animations like doors, trunks, roof latches, etc
# Multiplayer sync is automatic (networked by the addon): you don't need to do anything for it.


## Same as the main system, define the baseclass of the vehicle
```

DEFINE_BASECLASS( "base_glide_car" ) -- strictly necessary if you wanna make support for custom anims (for example speedometers or tachometers)

```
## Custom config parameters
### Set them before the 'if CLIENT' section
### NOTE: this config is read ONCE per vehicle and cached. If you ever change
### it at runtime, call BlackterioCustomAnims.InvalidateConfig( self ) after.
```
-- Custom config for doors animations
function ENT:GetAnimationConfig()
    return {
        count = 2, -- Door Count (9 max)
        names = { "doorfront", "doorback" }, -- pose parameter names, IN ORDER

        -- Sounds: omit these two keys to use the addon's default bus door
        -- sounds. Use "" to disable a sound entirely.
        initialSound = "blackterios_glide_vehicles/doors/caropen.wav",
        finishSound  = "blackterios_glide_vehicles/doors/carclose.wav",

        duration = 0.8, -- Animation duration IN SECONDS. It is exact and the
                        -- same on every client, at any FPS.
        lerpType = "linear", -- Lerp type: "linear", "smooth", "fast_start", "slow_start".
                             -- If omitted, "smooth" is used. Unknown values fall back to "linear".
        soundFollowsDuration = false -- set this to true if you want the sounds to play exactly when the animation value reach 1 or 0, useful for car doors, trunks, hoods, etc
    }
end
```
## Advanced custom config parameters
## Also set them before the 'if CLIENT' section
### Used to customize specific pose parameters
```
function ENT:GetAdvancedAnimationConfig()
    return {
        {
        name = "doorfront",
        customInitialSound = "vehicles/door_open.wav",
        duration = 0.4,
        lerpType = "smooth",
        soundFollowsDuration = true
        },
        {
        name = "doorback",
        customInitialSound = "vehicles/trunk_open.wav",
        customFinishSound = "vehicles/trunk_close.wav"
        },
        {
        name = "backwiper",
        customInitialSound = "vehicles/wiperstart.wav",
        customFinishSound = "vehicles/wiperfinish.wav",
        isDoor = false -- animation won't play when "play all animations" key is pressed (numpad 0 by default)
        },
    }
end
```

## Indicate that your vehicle has the animation system set
### Set this in `function ENT:OnUpdateAnimations()`
### IMPORTANT: if your vehicle does NOT use the main extra functions system
### (BlackterioExtraFunctions:UpdateAnimations), also call the BaseClass so
### the Glide base animations (steering wheel, etc.) keep working:

```
function ENT:OnUpdateAnimations()
    BaseClass.OnUpdateAnimations( self ) -- not needed if BlackterioExtraFunctions:UpdateAnimations() is called here too (it already calls the base)

    -- Custom animations (doors, panels, etc.)
    if BlackterioCustomAnims then
        BlackterioCustomAnims.UpdateAnimations( self )
    end
end
```

## Optional: query a door's state from your own code (works on server and client)

```
    local isOpen = BlackterioCustomAnims.GetAnimationState( self, 1 ) -- animation index 1-9
```
